    <script>
        // 全局游戏数据
        const gameData = {
            money: 1500,
            population: 0,
            power: 0,
            water: 0,
            happy: 80,
            tax: 0,
            currentBuild: null,
            mapData: []
        };
        let tileList = [];
        const ROW = 28;
        const COL = 45;

        // 初始化地图
        function initMap() {
            const mapDom = document.getElementById('map');
            mapDom.innerHTML = '';
            tileList = [];
            gameData.mapData = [];

            for(let r = 0; r < ROW; r++){
                gameData.mapData[r] = [];
                for(let c = 0; c < COL; c++){
                    const tile = document.createElement('div');
                    tile.className = 'tile';
                    tile.dataset.row = r;
                    tile.dataset.col = c;
                    tile.dataset.type = '';
                    mapDom.appendChild(tile);
                    tileList.push(tile);
                    gameData.mapData[r][c] = '';

                    // 格子点击事件
                    tile.onclick = function(){
                        buildTile(this);
                    };
                }
            }
        }

        // 选择建造项
        const buildBtns = document.querySelectorAll('.build-btn');
        buildBtns.forEach(btn => {
            btn.onclick = function(){
                buildBtns.forEach(b => b.classList.remove('active'));
                this.classList.add('active');
                gameData.currentBuild = {
                    type: this.dataset.type,
                    cost: Number(this.dataset.cost)
                };
            };
        });

        // 放置建筑逻辑
        function buildTile(tile) {
            const build = gameData.currentBuild;
            if(!build) return showMsg("请先选择建造项目！");

            const r = Number(tile.dataset.row);
            const c = Number(tile.dataset.col);
            const oldType = tile.dataset.type;

            if(oldType) return showMsg("该位置已有建筑！");
            if(gameData.money < build.cost) return showMsg("金币不足！", true);

            // 扣费 + 刷新格子
            gameData.money -= build.cost;
            tile.classList.add(build.type);
            tile.dataset.type = build.type;
            gameData.mapData[r][c] = build.type;

            calcCityEffect();
            updateUI();
        }

        // 计算城市各项收益/人口/满意度/税收
        function calcCityEffect() {
            let roadNum = 0, houseNum = 0, factoryNum = 0;
            let parkNum = 0, powerNum = 0, waterNum = 0;

            // 统计各类建筑数量
            for(let r = 0; r < ROW; r++){
                for(let c = 0; c < COL; c++){
                    const t = gameData.mapData[r][c];
                    switch(t){
                        case 'road': roadNum++; break;
                        case 'house': houseNum++; break;
                        case 'factory': factoryNum++; break;
                        case 'park': parkNum++; break;
                        case 'power': powerNum++; break;
                        case 'water': waterNum++; break;
                    }
                }
            }

            // 基础资源
            gameData.power = powerNum * 120;
            gameData.water = waterNum * 120;

            // 人口判定：必须有电有水 + 道路连通
            if(gameData.power > 0 && gameData.water > 0 && roadNum > 0){
                gameData.population = houseNum * 12 + factoryNum * 6;
            }else{
                gameData.population = 0;
            }

            // 满意度：公园加分、工厂减分
            let baseHappy = 70;
            baseHappy += parkNum * 4;
            baseHappy -= factoryNum * 2;
            gameData.happy = Math.max(10, Math.min(100, baseHappy));

            // 税收：人口 + 工厂产出
            gameData.tax = Math.floor(gameData.population * 0.8 + factoryNum * 15);
        }

        // 自动税收结算（10秒一轮）
        setInterval(() => {
            if(gameData.population > 0){
                gameData.money += gameData.tax;
                showMsg(`税收结算：+${gameData.tax} 金币`);
            }
            updateUI();
        }, 10000);

        // ========== 已修改：灾害间隔改为 30分钟 = 1800000 毫秒 ==========
        setInterval(() => {
            const random = Math.random();
            if(random < 0.25){ // 25%概率触发灾害
                const idx = Math.floor(Math.random() * tileList.length);
                const target = tileList[idx];
                const r = Number(target.dataset.row);
                const c = Number(target.dataset.col);
                const type = target.dataset.type;

                if(type && type !== 'road'){
                    target.classList.add('disaster');
                    showMsg("⚠️ 突发灾害，建筑受损！", true);
                    setTimeout(() => {
                        target.className = 'tile';
                        target.dataset.type = '';
                        gameData.mapData[r][c] = '';
                        calcCityEffect();
                        updateUI();
                    }, 1200);
                }
            }
        }, 1800000);

        // 界面刷新
        function updateUI() {
            document.getElementById('money').innerText = gameData.money;
            document.getElementById('pop').innerText = gameData.population;
            document.getElementById('power').innerText = gameData.power;
            document.getElementById('water').innerText = gameData.water;
            document.getElementById('happy').innerText = gameData.happy;
            document.getElementById('tax').innerText = gameData.tax;

            // 满意度变色
            const happyBar = document.getElementById('happyBar');
            if(gameData.happy < 40){
                happyBar.className = 'status-item danger';
            }else{
                happyBar.className = 'status-item happy';
            }
        }

        // 消息提示
        function showMsg(text, isWarn = false) {
            const msg = document.getElementById('msg');
            msg.innerText = text;
            msg.style.opacity = 1;
            if(isWarn) msg.style.background = 'rgba(220,38,38,0.9)';
            else msg.style.background = 'rgba(0,0,0,0.85)';

            setTimeout(() => {
                msg.style.opacity = 0;
            }, 1800);
        }

        // 存档（本地浏览器存储）
        function saveGame() {
            localStorage.setItem('citySave', JSON.stringify(gameData));
            showMsg("✅ 城市存档成功！");
        }

        // 读档
        function loadGame() {
            const save = localStorage.getItem('citySave');
            if(!save) return showMsg("暂无存档记录！", true);

            const oldData = JSON.parse(save);
            Object.assign(gameData, oldData);

            // 还原地图样式
            for(let r = 0; r < ROW; r++){
                for(let c = 0; c < COL; c++){
                    const idx = r * COL + c;
                    const t = gameData.mapData[r][c];
                    tileList[idx].className = 'tile';
                    tileList[idx].dataset.type = t;
                    if(t) tileList[idx].classList.add(t);
                }
            }
            updateUI();
            showMsg("✅ 读取存档成功！");
        }

        // 重置游戏
        function resetGame() {
            if(!confirm("确定要清空整个城市，重新开始吗？")) return;
            gameData.money = 1500;
            gameData.population = 0;
            gameData.power = 0;
            gameData.water = 0;
            gameData.happy = 80;
            gameData.tax = 0;
            gameData.currentBuild = null;
            initMap();
            updateUI();
            showMsg("🔄 城市已重置！");
        }

        // 页面加载初始化
        window.onload = function() {
            initMap();
            updateUI();
        };
    </script>
